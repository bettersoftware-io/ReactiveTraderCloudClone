const APP = "http://localhost:3210";
// The script lives in the output directory — resolve OUT from its own URL so
// the run command's cwd (pnpm --filter sets it to packages/client-react)
// cannot misplace the screenshots.
const OUT = new URL(".", import.meta.url).pathname;

interface Shot {
  skin: string;
  label: string;
}

const SHOTS: Shot[] = [
  { skin: "holo3d", label: "Holo HUD 3D" },
  { skin: "terminal3d", label: "Terminal 3D" },
];

async function main(): Promise<void> {
  const browser = await chromium.launch();
  const page = await browser.newPage({
    viewport: { width: 1600, height: 900 },
  });
  await page.goto(APP);
  await page.getByTestId("tile-EURUSD").waitFor({ timeout: 30_000 });

  for (const shot of SHOTS) {
    await page.getByTestId("skin-picker").click();
    await page.getByText(shot.label, { exact: true }).click();
    await page.waitForTimeout(800); // let aurora/panels settle
    await page.screenshot({ path: `${OUT}/app-${shot.skin}-dark.png` });

    await page.getByTestId("theme-toggle").click(); // dark → light
    await page.waitForTimeout(800);
    await page.screenshot({ path: `${OUT}/app-${shot.skin}-light.png` });

    await page.getByTestId("theme-toggle").click(); // light → system
    await page.getByTestId("theme-toggle").click(); // system → dark
  }

  await browser.close();
}

await main();
